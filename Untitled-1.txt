        function normalizeUkPostcodeDisplay(raw) {
            var c = compactUkPostcode(raw);
            if (c === 'GIR0AA') return 'GIR 0AA';
            if (c.length < 5) return String(raw || '').trim().toUpperCase();
            return c.slice(0, -3) + ' ' + c.slice(-3);
        }
        function isValidUkPostcode(raw) {
            var c = compactUkPostcode(raw);
            if (c === 'GIR0AA') return true;
            if (c.length < 5 || c.length > 7) return false;
            return /^[A-Z]{1,2}[0-9][0-9A-Z]?[0-9][A-Z]{2}$/.test(c);
        }
        function normalizeTitleNumberHero(raw) {
            return String(raw || '').replace(/[\s\-]+/g, '').toUpperCase();
        }
        function isValidHmTitleNumberHero(raw) {
            var n = normalizeTitleNumberHero(raw);
            if (!n) return false;
            return /^[A-Z]{2,3}\d{3,12}$/.test(n);
        }
        function clearPostcodeSuggestions() {
            if (!postcodeSuggestions) return;
            postcodeSuggestions.classList.add('d-none');

        }
        function splitDoogalAddress(addressText, fallbackCity) {
            var parts = String(addressText || '').split(',').map(function (part) { return part.trim(); }).filter(Boolean);
            var line1 = '';
            var line2 = '';
            var city = fallbackCity || '';
            if (parts.length > 0) line1 = parts[0];
            if (parts.length > 2) {
                line2 = parts.slice(1, parts.length - 1).join(', ');
                city = parts[parts.length - 1];
            } else if (parts.length === 2) {
                city = parts[1];
            }
            return { line1: line1, line2: line2, city: city };

        function renderPostcodeSuggestions() {

            var start = heroAddressPage * heroAddressPageSize;

                postcodePager.classList.remove('d-none');
            }
            if (postcodePrev) {
                postcodePrev.disabled = heroAddressPage <= 0;
            }
            if (postcodeNext) {
                postcodeNext.disabled = end >= heroKnownAddresses.length;
            }

            }

        if (postcodePrev) {

                if (heroAddressPage <= 0) return;
                heroAddressPage -= 1;
                renderPostcodeSuggestions();
            });
        }
        if (postcodeNext) {
            postcodeNext.addEventListener('click', function () {
                var maxPage = Math.floor((heroKnownAddresses.length - 1) / heroAddressPageSize);
                if (heroAddressPage >= maxPage) return;
                heroAddressPage += 1;
                renderPostcodeSuggestions();
            });

            modeFields.forEach(function (field) {
                var active = field.getAttribute('data-mode') === mode;
                field.classList.toggle('active', active);

                    if (postcodeFeedback) postcodeFeedback.textContent = 'Enter a complete valid UK postcode.';
                    postcodeInput.focus();

                var displayPc = normalizeUkPostcodeDisplay(rawPc);
                postcodeInput.value = displayPc;
                if (postcodeFeedback) postcodeFeedback.textContent = 'Looking up postcode details...';
                var lookupUrl = postcodeApiUrl + '?postcode=' + encodeURIComponent(compactUkPostcode(displayPc));
